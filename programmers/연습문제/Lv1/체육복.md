# 체육복







```
def solution(n, lost, reserve):
    answer = 0
    len_lost = len(lost)
    len_reserve = len(reserve)
    
    first_time_possible = n - len_lost
    
    for i in range(1, n + 1):
        if len_lost == 0:
            answer += n
        if (i in lost) or (i in reserve):
            if (i in lost) and (i in reserve):
                if len_reserve == 1 and len_lost == 1:
                    answer += n
                if len_lost > 1 and len_reserve == 1:
                    answer = first_time_possible + 1
                # if len_lost > 1 and len_reserve > 1:
                #     if (not (i - 1)) in reserve or (not (i + 1)) in reserve:
                        
        if (i in lost) and i not in reserve:
            if ((i - 1) in reserve or (i + 1) in reserve) and len_reserve != 1:
                first_time_possible += 1
                answer = first_time_possible
            if len_reserve == 1:
                if (i - 1) in reserve or (i + 1) in reserve:
                    answer = n - 1
                    
    return answer
```
