```
def solution(arr1, arr2):
    answer = []
    
    max_arr = max(len(arr1), len(arr2))
    
    for i in range(max_arr):
        temp = []
        for j in range(len(arr2[0])):
            plus = 0
            for k in range(len(arr2)):
                plus += arr1[i][k] * arr2[k][j]
            temp.append(plus)
        answer.append(temp)
    
    return answer
```
