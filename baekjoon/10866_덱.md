import sys

N = int(sys.stdin.readline())

deq = []
for _ in range(N):
  a = sys.stdin.readline().split()
  if a[0] == 'push_back':
    deq.append(a[1])
  elif a[0] == 'push_front':
    deq = [a[1]] + deq
  elif a[0] == 'pop_front':
    if deq:
      print(deq.pop(0))
    else:
      print(-1)
  elif a[0] == 'pop_back':
    if deq:
      print(deq.pop())
    else:
      print(-1)
  elif a[0] == 'size':
    print(len(deq))
  elif a[0] == 'empty':
    if deq:
      print(0)
    else:
      print(1)
  elif a[0] == 'front':
    if deq:
      print(deq[0])
    else:
      print(-1)
  elif a[0] == 'back':
    if deq:
      print(deq[-1])
    else:
      print(-1)
