from collections import deque
import sys


N = int(sys.stdin.readline())

deque_ls = deque([])
for _ in range(N):
  ls = sys.stdin.readline().split()
  if len(ls) > 1:
    deque_ls.append(ls[1])
  elif ls[0] == "front":
    if deque_ls:
      print(deque_ls[0])
    else:
      print(-1)
  elif ls[0] == "size":
    print(len(deque_ls))
  elif ls[0] == "pop":
    if deque_ls:
      a = deque_ls.popleft()
      print(a)
    else:
      print(-1)
  elif ls[0] == "empty":
    if deque_ls:
      print(0)
    else:
      print(1)
  elif ls[0] == "back":
    if deque_ls:
      print(deque_ls[-1])
    else:
      print(-1)
